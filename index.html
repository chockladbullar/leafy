    <script>

        const DB = {

            key: 'leafy_v3_core',

            defaults: { plants: [], logs: [], settings: { mode: 'light', notifications: false } },

            data: null,

            init() {

                const saved = localStorage.getItem(this.key);

                this.data = saved ? JSON.parse(saved) : this.defaults;

                // Merge in case of missing keys from older backups

                this.data = { ...this.defaults, ...this.data };

                this.data.settings = { ...this.defaults.settings, ...this.data.settings };

                

                this.apply();

                this.checkNotifCapability();

            },

            save() {

                localStorage.setItem(this.key, JSON.stringify(this.data));

                render();

            },

            apply() {

                document.documentElement.classList.toggle('dark-mode', this.data.settings.mode === 'dark');

                const btn = document.getElementById('notif-btn');

                if(btn) {

                    btn.innerText = this.data.settings.notifications ? 'Active' : 'Setup';

                    btn.classList.toggle('active', !!this.data.settings.notifications);

                }

                const lBtn = document.getElementById('mode-light');

                const dBtn = document.getElementById('mode-dark');

                if(lBtn) lBtn.classList.toggle('active', this.data.settings.mode === 'light');

                if(dBtn) dBtn.classList.toggle('active', this.data.settings.mode === 'dark');

            },

            checkNotifCapability() {

                const isStandalone = window.navigator.standalone || window.matchMedia('(display-mode: standalone)').matches;

                const warning = document.getElementById('ios-notif-warning');

                if (!isStandalone && /iPhone|iPad|iPod/.test(navigator.userAgent)) {

                    if(warning) warning.style.display = 'block';

                }

            },

            reset() {

                localStorage.clear(); 

                location.reload();

            }

        };

        // Backup Export/Import

        function exportData() {

            const dataStr = JSON.stringify(DB.data, null, 2);

            const blob = new Blob([dataStr], { type: 'application/json' });

            const url = URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = `leafy_backup_${new Date().toISOString().split('T')[0]}.json`;

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            URL.revokeObjectURL(url);

            showToast("Backup exported!");

        }

        }


        function handleImport(event) {

            const file = event.target.files[0];

            if (!file) return;


            const reader = new FileReader();

            reader.onload = (e) => {

                try {

                    const importedData = JSON.parse(e.target.result);

                    if (importedData.plants) {

                        showModal("Overwrite Data?", "This will replace all current plants with the backup data.", () => {

                            const settings = { ...DB.defaults.settings, ...(importedData.settings || {}) };

                            DB.data = { ...DB.defaults, ...importedData, settings };

                            DB.save();

                            DB.apply();

                            showToast("Data imported!");

                            showPage('sanctuary');

                        });

                    } else {

                        showToast("Invalid backup file.");

                    }

                } catch (err) {

                    showToast("Error reading file.");

                }

                event.target.value = ''; 

            };

            reader.readAsText(file);

        }


        // Care Notifications

        async function toggleNotifications() {

            if (!("Notification" in window)) {

                showToast("Notifications not supported.");

                return;

            }

            if (Notification.permission === "granted") {

                DB.data.settings.notifications = !DB.data.settings.notifications;

                DB.save();

                DB.apply();

                showToast(DB.data.settings.notifications ? "Alerts enabled!" : "Alerts off.");

            } else {

                const permission = await Notification.requestPermission();

                if (permission === "granted") {

                    DB.data.settings.notifications = true;

                    DB.save();

                    DB.apply();

                    new Notification("Leafy", { body: "Care alerts are now active!" });

                }

            }

        }


        function checkPlantCare() {

            if (!DB.data.settings.notifications) return;

            const now = new Date();

            const thirsty = [];

            const hungry = [];


            DB.data.plants.forEach(p => {

                const hoursSinceWater = (now - new Date(p.lastWatered)) / 3600000;

                const hoursSinceFert = (now - new Date(p.lastFertilized || p.lastWatered)) / 3600000;

                

                if (hoursSinceWater > (p.waterInterval * 24)) thirsty.push(p.name);

                if (p.fertInterval && hoursSinceFert > (p.fertInterval * 24)) hungry.push(p.name);

            });


            if (thirsty.length) {

                const body = thirsty.length > 1 ? `${thirsty[0]} and ${thirsty.length - 1} others need water.` : `${thirsty[0]} needs water!`;

                sendNotification("Leafy Care", body);

            }

            if (hungry.length) {

                const body = hungry.length > 1 ? `${hungry[0]} and ${hungry.length - 1} others need nutrients.` : `${hungry[0]} needs fertilizing!`;

                sendNotification("Leafy Food", body);

            }

        }


        function sendNotification(title, body) {

            const lastNotif = sessionStorage.getItem('last_notif_' + title);

            const cooldown = 86400000; // 24 hour throttle

            if (!lastNotif || (Date.now() - lastNotif > cooldown)) { 

                new Notification(title, { body, icon: 'https://fonts.gstatic.com/s/i/short-term/release/materialsymbolsoutlined/potted_plant/default/192px.svg' });

                sessionStorage.setItem('last_notif_' + title, Date.now());

            }

        }

        function showPage(id) {

            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));

            const nav = document.getElementById('nav-'+id);

            if(nav) nav.classList.add('active');

            

            if(id === 'journal') renderJournal();

        }


        function savePlant() {

            const nameEl = document.getElementById('p-name');

            const waterEl = document.getElementById('p-water');

            const fertEl = document.getElementById('p-fert');

            

            const name = nameEl.value.trim();

            const water = parseInt(waterEl.value);

            const fert = parseInt(fertEl.value);

            

            if(!name) return showToast("Enter a name");

            if(water < 1) return showToast("Water interval must be at least 1 day");

            

            DB.data.plants.push({

                id: Date.now(), name, waterInterval: water, fertInterval: fert,

                lastWatered: new Date().toISOString(),

                lastFertilized: new Date().toISOString()

            });

            DB.save();

            

            nameEl.value = '';

            waterEl.value = '7';

            fertEl.value = '30';

            

            showPage('sanctuary');

            showToast("Plant added!");

        }


        function logCare(id, type) {

            const p = DB.data.plants.find(x => x.id === id);

            if(!p) return;

            const now = new Date().toISOString();

            if(type === 'water') p.lastWatered = now;

            else p.lastFertilized = now;

            

            DB.data.logs.push({ plant: p.name, type: type, date: now });

            DB.save();

            showToast(`${type === 'water' ? 'Watered' : 'Fertilized'} ${p.name}`);

        }


        function render() {

            const grid = document.getElementById('plant-grid');

            if(!grid) return;

            grid.innerHTML = '';

            document.getElementById('empty-sanctuary').style.display = DB.data.plants.length ? 'none' : 'block';

            document.getElementById('stat-total').innerText = DB.data.plants.length + " Plants";

            

            let thirstyCount = 0;

            DB.data.plants.forEach(p => {

                const daysSince = (new Date() - new Date(p.lastWatered)) / 86400000;

                const overdue = daysSince > p.waterInterval;

                if(overdue) thirstyCount++;

                const div = document.createElement('div');

                div.className = 'plant-card';

                div.innerHTML = `

                    <div style="height:120px; background:var(--surf-c); display:flex; align-items:center; justify-content:center;">

                        <span class="mi" style="font-size:3.5rem; color:var(--primary-dim); opacity:0.6;">potted_plant</span>

                    </div>

                    <div class="info">

                        <div style="font-weight:700; color:var(--primary); font-size:1.1rem; margin-bottom:4px;">${p.name}</div>

                        <div class="flex justify-between items-center mt1">

                            <span class="small uppercase" style="${overdue?'color:var(--error);font-weight:800;':'color:var(--outline);'}">${overdue?'Thirsty':'Healthy'}</span>

                            <div class="flex gap1">

                                <button class="care-btn" style="background:var(--ter-fixed-dim); color:var(--tertiary); width:38px; height:38px;" onclick="logCare(${p.id}, 'fert')">

                                    <span class="mi" style="font-size:18px;">nutrition</span>

                                </button>

                                <button class="care-btn" style="background:var(--sec-c); color:var(--primary); width:38px; height:38px;" onclick="logCare(${p.id}, 'water')">

                                    <span class="mi" style="font-size:18px;">water_drop</span>

                                </button>

                            </div>

                        </div>

                    </div>

                `;

                grid.appendChild(div);

            });

            const hydration = DB.data.plants.length ? Math.round(((DB.data.plants.length - thirstyCount) / DB.data.plants.length) * 100) : 100;

            document.getElementById('stat-hydration').innerText = hydration + "%";

        }

        // Environment Data

        async function fetchEnv() {

            if (!navigator.geolocation) return;

            navigator.geolocation.getCurrentPosition(async (pos) => {

                try {

                    const res = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${pos.coords.latitude}&longitude=${pos.coords.longitude}&current_weather=true&relative_humidity_2m=true&daily=sunrise,sunset&timezone=auto`);

                    const d = await res.json();

                    

                    const humEl = document.getElementById('env-humidity');

                    const srEl = document.getElementById('env-sunrise');

                    const ssEl = document.getElementById('env-sunset');

                    

                    if(humEl) humEl.innerText = (d.current_weather.relative_humidity_2m || '--') + "%";

                    if(srEl) srEl.innerText = new Date(d.daily.sunrise[0]).toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'});

                    if(ssEl) ssEl.innerText = new Date(d.daily.sunset[0]).toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'});

                    

                    updateSeasonInfo();

                } catch(e) {}

            });

        }


        function updateSeasonInfo() {

            const month = new Date().getMonth();

            const banner = document.getElementById('season-insight');

            const icon = document.getElementById('season-icon');

            const name = document.getElementById('season-name');

            const tip = document.getElementById('season-tip');

            

            if(!banner) return;

            banner.style.display = 'flex';

            

            if (month >= 2 && month <= 4) {

                name.innerText = "Spring Awakening"; icon.innerText = "local_florist";

                tip.innerText = "Plants are waking up! Increase watering and start fertilizing.";

            } else if (month >= 5 && month <= 8) {

                name.innerText = "Summer Growth"; icon.innerText = "wb_sunny";

                tip.innerText = "High growth phase. Keep soil moist and watch for heat stress.";

            } else if (month >= 9 && month <= 10) {

                name.innerText = "Autumn Slumber"; icon.innerText = "eco";

                tip.innerText = "Growth is slowing. Scale back watering as light levels drop.";

            } else {

                name.innerText = "Winter Dormancy"; icon.innerText = "ac_unit";

                tip.innerText = "Plants are resting. Water sparingly and stop fertilizing.";

            }

        }

            checkPlantCare(); 

        };
